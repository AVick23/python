class Reader:
    def __init__(self, name, age):
        self.name = name
        self.age = age
        self.borrowed_books = []
        
    def borrow_book(self, book):
        if book.available:
            book.available = False
            self.borrowed_books.append(book)
            return True
        return False
        
    def return_book(self, book):
        if book in self.borrowed_books:
            book.available = True
            self.borrowed_books.remove(book)
            return True
        return False
        
    def __str__(self):

        return f"{self.name} ({self.age} –ª–µ—Ç): {books_list}"
